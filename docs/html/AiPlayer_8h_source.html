89 int minimax(Board<T>* boardPtr, bool maximizingPlayer, T ai, T opp, int alpha, int beta, int depth, int N = 3) {
90 // Terminal conditions
91 // Check if AI (this player) wins
92 if (boardPtr->is_win(this)) return 1000 + depth; // Prefer faster wins
93
94 // Check if opponent wins - create temporary player with opponent symbol
95 Player<T>* tempOpponent = new Player<T>("temp_opp", opp, PlayerType::AI);
96 tempOpponent->set_board_ptr(boardPtr);
97 if (boardPtr->is_win(tempOpponent)) {
98 delete tempOpponent;
99 return -1000 - depth; // Prefer slower losses
100 }
101 delete tempOpponent;
102
103 if (boardPtr->is_draw(this)) return 0;
104 if (depth == 0) return evaluate_board(boardPtr, ai, opp, N);
105
106 int rows = boardPtr->get_rows();
107 int cols = boardPtr->get_columns();
108
109 if (maximizingPlayer) {
110 int best = numeric_limits<int>::min();
111 for (int i = 0; i < rows; i++) {
112 for (int j = 0; j < cols; j++) {
113 if (condition(i, j)) {
114 vector<T> moveValues = generate_move_values(i, j, true);
115 for (T moveVal : moveValues) {
116 Move<T>* move = createMove(i, j);
117 // Update move symbol if needed (for games with value-based moves)
118 if (move) {
119 boardPtr->update_board(move);
120 int val = minimax(boardPtr, false, ai, opp, alpha, beta, depth - 1, N);
121 Move<T>* undomove = createUndoMove(i, j);
122 if (undomove) {
123 boardPtr->update_board(undomove);
124 }
125 delete move;
126 delete undomove;
127
128 best = max(best, val);
129 alpha = max(alpha, best);
130 if (beta <= alpha) {
131 return best; // Alpha-beta pruning
132 }
133 }
134 }
135 }
136 }
137 }
138 return best;
139 } else {
140 int best = numeric_limits<int>::max();
141 for (int i = 0; i < rows; i++) {
142 for (int j = 0; j < cols; j++) {
143 if (condition(i, j)) {
144 vector<T> moveValues = generate_move_values(i, j, false);
145 for (T moveVal : moveValues) {
146 Move<T>* move = createMove(i, j);
147 if (move) {
148 boardPtr->update_board(move);
149 int val = minimax(boardPtr, true, ai, opp, alpha, beta, depth - 1, N);
150 Move<T>* undomove = createUndoMove(i, j);
151 if (undomove) {
152 boardPtr->update_board(undomove);
153 }
154 delete move;
155 delete undomove;
156
157 best = min(best, val);
158 beta = min(beta, best);
159 if (beta <= alpha) {
160 return best; // Alpha-beta pruning
161 }
162 }
163 }
164 }
165 }
166 }

238 virtual Move<T>* calculateMove() {
239 Board<T>* board_ptr = this->get_board_ptr();
240 if (!board_ptr) throw runtime_error("Board not assigned to player!");
241
242 int rows = board_ptr->get_rows();
243 int cols = board_ptr->get_columns();
244
245 int bestVal = numeric_limits<int>::min();
246 int bestX = -1, bestY = -1;
247
248 T ai = this->get_symbol();
249 T opp = get_opponent_symbol();
250
251 int maxDepth = get_max_depth();
252
253 for (int i = 0; i < rows; i++) {
254 for (int j = 0; j < cols; j++) {
255 if (condition(i, j)) {
256 vector<T> moveValues = generate_move_values(i, j, true);
257 for (T moveValue : moveValues) {
258 Move<T>* move = createMove(i, j);
259 if (!move) continue;
260
261 board_ptr->update_board(move);
262 int eval = minimax(board_ptr, false, ai, opp,
263 numeric_limits<int>::min(),
264 numeric_limits<int>::max(),
265 maxDepth);
266 Move<T>* undoMove = createUndoMove(i, j);
267 if (undoMove) {
268 board_ptr->update_board(undoMove);
269 delete undoMove;
270 }
271 delete move;
272
273 if (eval > bestVal) {
274 bestVal = eval;
275 bestX = i;
276 bestY = j;
277 }
278 }
279 }
280 }
281 }
282
283 if (bestX == -1 || bestY == -1) {
284 // Fallback: find first valid move
285 for (int i = 0; i < rows && bestX == -1; i++) {
286 for (int j = 0; j < cols && bestX == -1; j++) {
287 if (condition(i, j)) {
288 bestX = i;
289 bestY = j;
290 }
291 }
292 }
293 }
294
295 return createMove(bestX, bestY);
296 }
