266 GameManager(Board<T>* b, Player<T>* p[2], UI<T>* u)
267 : boardPtr(b), ui(u) {

272 }
273
277 void run() {
278 ui->display_board_matrix(boardPtr->get_board_matrix());
279 Player<T>* currentPlayer = players[0];
280
281 while (true) {
282 for (int i : {0, 1}) {
283 currentPlayer = players[i];
284 Move<T>* move = ui->get_move(currentPlayer);
285
286 while (!boardPtr->update_board(move))
287 move = ui->get_move(currentPlayer);
288
289 ui->display_board_matrix(boardPtr->get_board_matrix());
290
291 if (boardPtr->is_win(currentPlayer)) {
292 ui->display_message(currentPlayer->get_name() + " wins!");
293 return;
294 }
295 if (boardPtr->is_lose(currentPlayer)) {
296 ui->display_message(players[1 - i]->get_name() + " wins!");
297 return;
298 }
299 if (boardPtr->is_draw(currentPlayer)) {
300 ui->display_message("Draw!");
301 return;
302 }
303 }
304 }
305 }
