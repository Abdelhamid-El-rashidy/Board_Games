22bool Four_Four_XO_Board::update_board(Move<char> *move) {
23
24 dualMove* dmove = dynamic_cast<dualMove*>(move);
25 if (!dmove) return false;
26
27 int x = dmove->get_x();
28 int y = dmove->get_y();
29 int dy = dmove->get_dy();
30 int dx = dmove->get_dx();
31 char symbol = dmove->get_symbol();
32
33 if (x<0 || x>=rows || y<0 || y>=columns || board[x][y] != symbol || abs(dy+dx)!=1 ) return false;
34 if (x+dx<0 || x+dx>=rows || y+dy<0 || y+dy>=columns || board[x+dx][y+dy] != blank_symbol) return false;
35
36 n_moves++;
37 board[x][y] = blank_symbol;
38 board[x+dx][y+dy] = symbol;
39 return true;

42bool Four_Four_XO_Board::is_win(Player<char>* player) {
43 const char sym = player->get_symbol();
44
45 auto all_equal = [&](char a, char b, char c) {

48
49 // Check rows and columns
50 for (int i = 0; i < rows; ++i) {
51 if ((all_equal(board[i][0], board[i][1], board[i][2]) && board[i][0] == sym) ||
52 (all_equal(board[0][i], board[1][i], board[2][i]) && board[0][i] == sym))
53 return true;
