23
24 // Validate move and apply if valid

49bool Infinity_X_O_Board::is_win(Player<char>* player) {
50 const char sym = player->get_symbol();
51
52 auto all_equal = [&](char a, char b, char c) {
53 return a == b && b == c && a != blank_symbol;
54 };
55
56 // Check rows and columns
57 for (int i = 0; i < rows; ++i) {
58 if ((all_equal(board[i][0], board[i][1], board[i][2]) && board[i][0] == sym) ||
59 (all_equal(board[0][i], board[1][i], board[2][i]) && board[0][i] == sym))
60 return true;
61 }
62
63 // Check diagonals
64 if ((all_equal(board[0][0], board[1][1], board[2][2]) && board[1][1] == sym) ||
65 (all_equal(board[0][2], board[1][1], board[2][0]) && board[1][1] == sym))
66 return true;
67
68 return false;
69}
