115 int R = b.size();
116 int C = b[0].size();
117 int score = 0;
118
119 // Example scoring: +10 for correct letter in position, +3 for empty, -5 otherwise
120 for (int i = 0; i < R; i++)
121 for (int j = 0; j < C; j++) {
122 // Check horizontal word match possibility
123 if (j + word.size() <= C) {
124 int s = 0;
125 for (int k = 0; k < word.size(); k++) {
126 if (b[i][j+k] == word[k]) s += 10;

128 else { s = -50; break; }

130 score += s;
131 }

137 int minimax(vector<vector<char>>& b,
138 bool maximizingPlayer,
139 char ai,
140 char opp,
141 int alpha,
142 int beta,
143 int depth,
144 int N = 3)
145{
146 // node accounting
147 ++node_counter;
148 if (node_counter > MAX_NODES) {
149 // abort search early: return heuristic evaluation to let caller use partial result
150 return evaluate_board(b, ai, opp, N);
151 }
152
153 if (is_win(b, ai, N)) return 1000 + depth; // prefer faster win
154 if (is_win(b, opp, N)) return -1000 - depth; // prefer slower loss
155 if (is_draw(b, N)) return 0;
156 if (depth == 0) return evaluate_board(b, ai, opp, N);
157
158 int rows = b.size();
159 int cols = b[0].size();
160
161 if (maximizingPlayer) {
162 int best = numeric_limits<int>::min();
163 for (int i = 0; i < rows; ++i) {
164 for (int j = 0; j < cols; ++j) {
165 if (b[i][j] != '.') continue;
166 for (char c : possibleLetters) {
167 b[i][j] = c;
168 int val = minimax(b, false, ai, opp, alpha, beta, depth - 1, N);
169 b[i][j] = '.';
170 if (val > best) best = val;
171 if (best > alpha) alpha = best;
172 if (beta <= alpha) {
173 // pruning - break out early
174 goto OUT_MAX_LOOP;
175 }
176 // early node budget check
177 if (node_counter > MAX_NODES) goto OUT_MAX_LOOP;
178 }
179 }
180 }
181OUT_MAX_LOOP:
182 return best;
183 } else {
184 int best = numeric_limits<int>::max();
185 for (int i = 0; i < rows; ++i) {
186 for (int j = 0; j < cols; ++j) {
187 if (b[i][j] != '.') continue;
188 for (char c : possibleLetters) {
189 b[i][j] = c;
190 int val = minimax(b, true, ai, opp, alpha, beta, depth - 1, N);
191 b[i][j] = '.';
192 if (val < best) best = val;
193 if (best < beta) beta = best;
194 if (beta <= alpha) {
195 goto OUT_MIN_LOOP;
196 }
197 if (node_counter > MAX_NODES) goto OUT_MIN_LOOP;
198 }
199 }
200 }
201OUT_MIN_LOOP:
202 return best;
203 }

265void load_global_dictionary(const string& path = "../dic.txt") {
266 if (GLOBAL_DICT_LOADED) return;
267 ifstream f(path);
268 if (!f.is_open()) {
269 cerr << "Warning: cannot open " << path << "\n";
270 GLOBAL_DICT_LOADED = true; // prevent repeated warnings
271 return;
272 }
273 string line;
274 while (getline(f, line)) {
275 // trim
276 size_t s = 0, e = line.size();
277 while (s < e && isspace((unsigned char)line[s])) ++s;
278 while (e > s && isspace((unsigned char)line[e-1])) --e;
279 if (e - s == 0) continue;
280 string w = line.substr(s, e - s);
281 for (char &c : w) c = toupper(static_cast<unsigned char>(c));
282 GLOBAL_DICT.insert(w);
283 }
284 GLOBAL_DICT_LOADED = true;
285}
