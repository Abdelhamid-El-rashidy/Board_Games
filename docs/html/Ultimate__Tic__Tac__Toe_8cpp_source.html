30bool mainBoard::is_win(Player<char>* player) {
31 const char sym = player->get_symbol();
32
33 auto all_equal = [&](char a, char b, char c) {
34 return a == b && b == c && a != blank_symbol;
35 };
36
37 // Check rows and columns
38 for (int i = 0; i < rows; ++i) {
39 if ((all_equal(board[i][0], board[i][1], board[i][2]) && board[i][0] == sym) ||
40 (all_equal(board[0][i], board[1][i], board[2][i]) && board[0][i] == sym))
41 return true;
42 }
43
44 // Check diagonals
45 if ((all_equal(board[0][0], board[1][1], board[2][2]) && board[1][1] == sym) ||
46 (all_equal(board[0][2], board[1][1], board[2][0]) && board[1][1] == sym))
47 return true;

69bool Ultimate_XO_Board::update_board(Move<char>* move) {
70 int x = move->get_x();
71 int y = move->get_y();
72 char mark = move->get_symbol();
73
74 auto checkboard = [&](int dx, int dy) {
75 const char sym = move->get_symbol();
76
77 auto all_equal = [&](char a, char b, char c) {
78 return a == b && b == c && a != blank_symbol;
79 };
80
81 // Check rows and columns
82 for (int i = 0; i < 3; ++i) {
83 if ((all_equal(board[i+dx][0+dy], board[i+dx][1+dy], board[i+dx][2+dy]) && board[i+dx][0+dy] == sym) ||
84 (all_equal(board[0+dx][i+dy], board[1+dx][i+dy], board[2+dx][i+dy]) && board[0+dx][i+dy] == sym))
85 return true;
86 }
87
88 // Check diagonals
89 if ((all_equal(board[dx+0][dy+0], board[dx+1][dy+1], board[dx+2][dy+2]) && board[dx+1][dy+1] == sym) ||
90 (all_equal(board[dx+0][dy+2], board[dx+1][dy+1], board[dx+2][dy+0]) && board[dx+1][dy+1] == sym))
91 return true;
92
93 return false;
94 };
95
96 // Validate move and apply if valid
97 if (!(x < 0 || x >= rows || y < 0 || y >= columns) && (board[x][y] == blank_symbol)) {
98 board[x][y] = toupper(mark);
99 n_moves++;
100 if (checkboard((x/3) * 3,(y/3) * 3)) mainBoardPtr->update_board(move);
101 return true;
102 }
103 return false;
104}
