26 // Heuristic evaluation for large boards
27 virtual int evaluate_board(const vector<vector<T>>& b, T ai, T opp, int N = 3) const {
28 int score = 0;
29 int rows = b.size();
30 int cols = b[0].size();
31
32 auto count_sequence = [&](int i, int j, int dx, int dy, T symbol) {
33 int cnt = 0;
34 for (int k = 0; k < N; k++) {
35 int x = i + k * dx;
36 int y = j + k * dy;
37 if (x < 0 || x >= rows || y < 0 || y >= cols) return 0;
38 if (b[x][y] == symbol) cnt++;
39 else if (b[x][y] != '.') return 0; // blocked
40 }
41 return cnt;
42 };
43
44 for (int i = 0; i < rows; i++)
45 for (int j = 0; j < cols; j++) {
46 score += count_sequence(i, j, 0, 1, ai); // row
47 score += count_sequence(i, j, 1, 0, ai); // column
48 score += count_sequence(i, j, 1, 1, ai); // diag TL-BR
49 score += count_sequence(i, j, 1, -1, ai); // diag TR-BL
50
51 score -= count_sequence(i, j, 0, 1, opp);
52 score -= count_sequence(i, j, 1, 0, opp);
53 score -= count_sequence(i, j, 1, 1, opp);
54 score -= count_sequence(i, j, 1, -1, opp);
55 }
56
57 return score;
58 }

60 virtual bool is_win(const vector<vector<T>>& b, T symbol, int N = 3) const {
61 int rows = b.size();
62 int cols = b[0].size();
63
64 // Rows
65 for (int i = 0; i < rows; i++)
66 for (int j = 0; j <= cols - N; j++) {
67 bool win = true;
68 for (int k = 0; k < N; k++)
69 if (b[i][j + k] != symbol) { win = false; break; }
70 if (win) return true;
71 }
72
73 // Columns
74 for (int j = 0; j < cols; j++)
75 for (int i = 0; i <= rows - N; i++) {
76 bool win = true;
77 for (int k = 0; k < N; k++)
78 if (b[i + k][j] != symbol) { win = false; break; }
79 if (win) return true;
80 }
81
82 // Diagonal TL-BR
83 for (int i = 0; i <= rows - N; i++)
84 for (int j = 0; j <= cols - N; j++) {
85 bool win = true;
86 for (int k = 0; k < N; k++)
87 if (b[i + k][j + k] != symbol) { win = false; break; }
88 if (win) return true;
89 }
90
91 // Diagonal TR-BL
92 for (int i = 0; i <= rows - N; i++)
93 for (int j = N - 1; j < cols; j++) {
94 bool win = true;
95 for (int k = 0; k < N; k++)
96 if (b[i + k][j - k] != symbol) { win = false; break; }
97 if (win) return true;
98 }
99
100 return false;
101 }

103 virtual bool is_draw(const vector<vector<T>>& b, int N = 3) const {
104 for (auto &row : b)
105 for (auto &cell : row)
106 if (cell == '.') return false;
107 return !is_win(b, this->get_symbol(), N) && !is_win(b, get_opponent_symbol(), N);
108 }

110 // Minmax Alg with alpha bata optimization and depth optimization to avoid excessive search for large boards
111 // https://www.geeksforgeeks.org/dsa/minimax-algorithm-in-game-theory-set-4-alpha-beta-pruning/
112 virtual int minimax(vector<vector<T>>& b, bool maximizingPlayer, T ai, T opp, int alpha, int beta, int depth, int N = 3) {
113 if (is_win(b, ai, N)) return 1000 + depth; // prefer faster win
114 if (is_win(b, opp, N)) return -1000 - depth; // prefer slower loss
115 if (is_draw(b, N)) return 0;
116 if (depth == 0) return evaluate_board(b, ai, opp, N);
117
118 int rows = b.size();
119 int cols = b[0].size();
120
121 if (maximizingPlayer) {
122 int best = numeric_limits<int>::min();
123 for (int i = 0; i < rows; i++)
124 for (int j = 0; j < cols; j++)
125 if (b[i][j] == '.') {
126 b[i][j] = ai;
127 int val = minimax(b, false, ai, opp, alpha, beta, depth - 1, N);
128 b[i][j] = '.';
129 best = max(best, val);
130 alpha = max(alpha, best);
131 if (beta <= alpha) return best;
132 }
133 return best;
134 } else {
135 int best = numeric_limits<int>::max();
136 for (int i = 0; i < rows; i++)
137 for (int j = 0; j < cols; j++)
138 if (b[i][j] == '.') {
139 b[i][j] = opp;
140 int val = minimax(b, true, ai, opp, alpha, beta, depth - 1, N);
141 b[i][j] = '.';
142 best = min(best, val);
143 beta = min(beta, best);
144 if (beta <= alpha) return best;
145 }
146 return best;
147 }
148 }

153 virtual pair<int,int> calculateMove() {
154 Board<T>* board_ptr = this->get_board_ptr();
155 if (!board_ptr) throw runtime_error("Board not assigned to player!");
156
157 auto tempBoard = board_ptr->get_board_matrix();
158 int rows = tempBoard.size();
159 int cols = tempBoard[0].size();
160
161 int bestVal = numeric_limits<int>::min();
162 int bestX = -1, bestY = -1;
163
164 T ai = this->get_symbol();
165 T opp = get_opponent_symbol();
166
167 int maxDepth = 3; // tweak: higher depth = stronger AI but slower
168
169 for (int i = 0; i < rows; i++)
170 for (int j = 0; j < cols; j++)
171 if (tempBoard[i][j] == '.') {
172 tempBoard[i][j] = ai;
173 int moveVal = minimax(tempBoard, false, ai, opp,
174 numeric_limits<int>::min(),
175 numeric_limits<int>::max(),
176 maxDepth);
177 tempBoard[i][j] = '.';
178 if (moveVal > bestVal) {
179 bestVal = moveVal;
180 bestX = i;
181 bestY = j;
182 }
183 }
184
185 return {bestX, bestY};
186 }
