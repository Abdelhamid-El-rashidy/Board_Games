131 int minimax(Board<char>* boardPtr, bool maximizingPlayer, char ai, char opp, int alpha, int beta, int depth, int N = 3) {
132 // Terminal conditions
133 if (boardPtr->is_win(this)) return 1000 + depth;
134
135 Player<char>* tempOpp = new Player<char>("temp", opp, PlayerType::AI);
136 tempOpp->set_board_ptr(boardPtr);
137 if (boardPtr->is_win(tempOpp)) {
138 delete tempOpp;
139 return -1000 - depth;
140 }
141 delete tempOpp;
142
143 if (boardPtr->is_draw(this)) return 0;
144 if (depth == 0) return evaluate_board(boardPtr, ai, opp, N);
145
146 int rows = boardPtr->get_rows();
147 int cols = boardPtr->get_columns();
148
149 if (maximizingPlayer) {
150 int best = numeric_limits<int>::min();
151 for (int i = 0; i < rows; i++) {
152 for (int j = 0; j < cols; j++) {
153 // Check for AI's pieces
154 if (!condition_for_symbol(i, j, ai)) continue;
155
156 for (auto& dir : directions) {
157 int dx = dir.first;
158 int dy = dir.second;
159 if (i + dx < 0 || i + dx >= rows || j + dy < 0 || j + dy >= cols) continue;
160 if (boardPtr->get_cell(i + dx, j + dy) != blank_symbol) continue;
161
162 // Try the move
163 Move<char>* move = new dualMove(i, j, ai, dy, dx);
164 if (!move) continue;
165
166 if (boardPtr->update_board(move)) {
167 int val = minimax(boardPtr, false, ai, opp, alpha, beta, depth - 1, N);
168
169 // Undo the move - move piece back from (i+dx, j+dy) to (i, j)
170 Move<char>* undomove = createUndoMoveForSymbol(i, j, dx, dy, ai);
171 if (undomove && boardPtr->update_board(undomove)) {
172 delete undomove;
173 } else {
174 // Undo failed - this should never happen but handle it
175 delete undomove;
176 }
177 delete move;
178
179 best = max(best, val);
180 alpha = max(alpha, best);
181 if (beta <= alpha) {
182 return best;
183 }
184 } else {
185 delete move;
186 }
187 }
188 }
189 }
190 return best;
191 } else {
192 int best = numeric_limits<int>::max();
193 for (int i = 0; i < rows; i++) {
194 for (int j = 0; j < cols; j++) {
195 // Check for opponent's pieces
196 if (!condition_for_symbol(i, j, opp)) continue;
197
198 for (auto& dir : directions) {
199 int dx = dir.first;
200 int dy = dir.second;
201 if (i + dx < 0 || i + dx >= rows || j + dy < 0 || j + dy >= cols) continue;
202 if (boardPtr->get_cell(i + dx, j + dy) != blank_symbol) continue;
203
204 // Create move with opponent's symbol
205 Move<char>* move = new dualMove(i, j, opp, dy, dx);
206 if (!move) continue;
207
208 if (boardPtr->update_board(move)) {
209 int val = minimax(boardPtr, true, ai, opp, alpha, beta, depth - 1, N);
210
211 // Undo the move - move piece back from (i+dx, j+dy) to (i, j)
212 Move<char>* undomove = createUndoMoveForSymbol(i, j, dx, dy, opp);
213 if (undomove && boardPtr->update_board(undomove)) {
214 delete undomove;
215 } else {
216 // Undo failed - this should never happen but handle it
217 delete undomove;
218 }
219 delete move;
220
221 best = min(best, val);
222 beta = min(beta, best);
223 if (beta <= alpha) {
224 return best;
225 }
226 } else {
227 delete move;
228 }
229 }
230 }
231 }
232 return best;

239 virtual int evaluate_board(Board<char>* boardPtr, char ai, char opp, int N = 3) override {
240 if (boardPtr->is_win(this)) return 1000;
241
242 Player<char>* tempOpp = new Player<char>("temp", opp, PlayerType::AI);
243 tempOpp->set_board_ptr(boardPtr);
244 if (boardPtr->is_win(tempOpp)) {
245 delete tempOpp;
246 return -1000;
247 }
248 delete tempOpp;
249
250 // Simple heuristic: count pieces in winning positions
251 // Check for 2 in a row (potential win)
252 int score = 0;
253 auto board = boardPtr->get_board_matrix();
254 int rows = board.size();
255 int cols = board[0].size();
256 char blank = '.';
257
258 // Check for potential winning lines
259 auto count_line = [&](int x, int y, int dx, int dy, char sym) -> int {
260 int count = 0;
261 for (int k = 0; k < 3; k++) {
262 int nx = x + k * dx;
263 int ny = y + k * dy;
264 if (nx >= 0 && nx < rows && ny >= 0 && ny < cols) {
265 if (board[nx][ny] == sym) count++;
266 else if (board[nx][ny] != blank) return -1; // blocked
267 }
268 }
269 return count;
270 };
271
272 for (int i = 0; i < rows; i++) {
273 for (int j = 0; j < cols; j++) {
274 // Check horizontal
275 int ai_count = count_line(i, j, 0, 1, ai);
276 int opp_count = count_line(i, j, 0, 1, opp);
277 if (ai_count > 0) score += ai_count * 10;
278 if (opp_count > 0) score -= opp_count * 10;
279
280 // Check vertical
281 ai_count = count_line(i, j, 1, 0, ai);
282 opp_count = count_line(i, j, 1, 0, opp);
283 if (ai_count > 0) score += ai_count * 10;
284 if (opp_count > 0) score -= opp_count * 10;
285
286 // Check diagonal
287 ai_count = count_line(i, j, 1, 1, ai);
288 opp_count = count_line(i, j, 1, 1, opp);
289 if (ai_count > 0) score += ai_count * 10;
290 if (opp_count > 0) score -= opp_count * 10;
291 }
292 }
293
294 return score;
295 }

306 virtual Move<char>* calculateMove() override {
307 Board<char>* board_ptr = this->get_board_ptr();
308 if (!board_ptr) throw runtime_error("Board not assigned to player!");
309
310 int rows = board_ptr->get_rows();
311 int cols = board_ptr->get_columns();
312
313 int bestVal = numeric_limits<int>::min();
314 int bestX = -1, bestY = -1, bestDx = 0, bestDy = 0;
315
316 char ai = this->get_symbol();
317 char opp = get_opponent_symbol();
318
319 int maxDepth = get_max_depth();
320
321 // Iterate over all positions
322 for (int i = 0; i < rows; i++) {
323 for (int j = 0; j < cols; j++) {
324 // Check if position has player's piece
325 if (!condition(i, j)) continue;
326
327 // Try each direction
328 for (auto& dir : directions) {
329 int dx = dir.first;
330 int dy = dir.second;
331
332 // Check if move is valid
333 if (i + dx < 0 || i + dx >= rows || j + dy < 0 || j + dy >= cols) continue;
334 if (board_ptr->get_cell(i + dx, j + dy) != blank_symbol) continue;
335
336 // Try the move
337 current_dx = dx;
338 current_dy = dy;
339 Move<char>* move = createMove(i, j);
340 if (!move) continue;
341
342 if (board_ptr->update_board(move)) {
343 int eval = minimax(board_ptr, false, ai, opp,
344 numeric_limits<int>::min(),
345 numeric_limits<int>::max(),
346 maxDepth);
347
348 // Undo the move - move piece back from (i+dx, j+dy) to (i, j)
349 Move<char>* undoMove = createUndoMoveForSymbol(i, j, dx, dy, ai);
350 if (undoMove && board_ptr->update_board(undoMove)) {
351 delete undoMove;
352 } else {
353 delete undoMove;
354 }
355 delete move;
356
357 if (eval > bestVal) {
358 bestVal = eval;
359 bestX = i;
360 bestY = j;
361 bestDx = dx;
362 bestDy = dy;
363 }
364 } else {
365 delete move;
366 }
367 }
368 }
369 }
370
371 if (bestX == -1 || bestY == -1) {
372 // Fallback: find first valid move
373 for (int i = 0; i < rows && bestX == -1; i++) {
374 for (int j = 0; j < cols && bestX == -1; j++) {
375 if (condition(i, j)) {
376 for (auto& dir : directions) {
377 int dx = dir.first;
378 int dy = dir.second;
379 if (i + dx >= 0 && i + dx < rows && j + dy >= 0 && j + dy < cols) {
380 if (board_ptr->get_cell(i + dx, j + dy) == blank_symbol) {
381 bestX = i;
382 bestY = j;
383 bestDx = dx;
384 bestDy = dy;
385 break;
386 }
387 }
388 }
389 }
390 }
391 }
392 }
393
394 current_dx = bestDx;
395 current_dy = bestDy;
396 return createMove(bestX, bestY);
397 }
