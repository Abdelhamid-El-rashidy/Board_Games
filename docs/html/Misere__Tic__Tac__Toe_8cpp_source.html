20bool Misere_Tic_Tac_Toe_board::update_board(Move<char>* move) {
21 int x = move->get_x();
22 int y = move->get_y();
23 char mark = move->get_symbol();
24
25 // Validate move and apply if valid
26 if (!(x < 0 || x >= rows || y < 0 || y >= columns) &&
27 (board[x][y] == blank_symbol || mark == 0)) {
28
29 if (mark == 0) { // Undo move
30 n_moves--;
31 board[x][y] = blank_symbol;
32 }
33 else { // Apply move
34 n_moves++;
35 board[x][y] = toupper(mark);
36 }
37 return true;
38 }
39 return false;
40}
41bool Misere_Tic_Tac_Toe_board::is_lose(Player<char>* player) {
42 const char sym = player->get_symbol();
43
44 auto all_equal = [&](char a, char b, char c) {
45 return a == b && b == c && a != blank_symbol;
46 };
47
48 // Check rows and columns
49 for (int i = 0; i < rows; ++i) {
50 if ((all_equal(board[i][0], board[i][1], board[i][2]) && board[i][0] == sym) ||
51 (all_equal(board[0][i], board[1][i], board[2][i]) && board[0][i] == sym))
52 return true;
53 }
54
55 // Check diagonals
56 if ((all_equal(board[0][0], board[1][1], board[2][2]) && board[1][1] == sym) ||
57 (all_equal(board[0][2], board[1][1], board[2][0]) && board[1][1] == sym))
58 return true;
59
60 return false;
61}
