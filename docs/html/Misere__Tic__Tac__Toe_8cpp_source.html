21bool Misere_Tic_Tac_Toe_board::update_board(Move<char>* move) {
22 int x = move->get_x();
23 int y = move->get_y();
24 char mark = move->get_symbol();
25
26 // Validate move and apply if valid
27 if (!(x < 0 || x >= rows || y < 0 || y >= columns) &&
28 (board[x][y] == blank_symbol || mark == 0)) {
29
30 if (mark == 0) { // Undo move
31 n_moves--;
32 board[x][y] = blank_symbol;
33 }
34 else { // Apply move
35 n_moves++;
36 board[x][y] = toupper(mark);
37 }
38 return true;
39 }
40 return false;
41}
42bool Misere_Tic_Tac_Toe_board::is_lose(Player<char>* player) {
43 const char sym = player->get_symbol();
44
45 auto all_equal = [&](char a, char b, char c) {
46 return a == b && b == c && a != blank_symbol;
47 };
48
49 // Check rows and columns
50 for (int i = 0; i < rows; ++i) {
51 if ((all_equal(board[i][0], board[i][1], board[i][2]) && board[i][0] == sym) ||
52 (all_equal(board[0][i], board[1][i], board[2][i]) && board[0][i] == sym))
53 return true;
54 }
55
56 // Check diagonals
57 if ((all_equal(board[0][0], board[1][1], board[2][2]) && board[1][1] == sym) ||
58 (all_equal(board[0][2], board[1][1], board[2][0]) && board[1][1] == sym))
59 return true;
60
61 return false;
62}

76bool MisereSmartPlayer::check_3_in_row(const vector<vector<char>>& b, char symbol) const {
77 for (int i = 0; i < 3; i++) {
78 if (b[i][0] == symbol && b[i][1] == symbol && b[i][2] == symbol) return true;
79 if (b[0][i] == symbol && b[1][i] == symbol && b[2][i] == symbol) return true;
80 }
81 if (b[0][0] == symbol && b[1][1] == symbol && b[2][2] == symbol) return true;
82 if (b[0][2] == symbol && b[1][1] == symbol && b[2][0] == symbol) return true;
83 return false;
84}

86int MisereSmartPlayer::minimax_misere(vector<vector<char>>& b, bool isMaximizing, int depth) {
87 char ai = this->symbol;
88 char opp = get_opponent_symbol();
89
90 if (check_3_in_row(b, ai)) return -10 + depth;
91
92 if (check_3_in_row(b, opp)) return 10 - depth;
93
94
95 bool full = true;
96 for(auto& r : b) for(auto c : r) if(c == '.') full = false;
97 if(full) return 0;
98
99 // 2. Recursion
100 if (isMaximizing) {
101 int best = -1000;
102 for (int i = 0; i < 3; i++) {
103 for (int j = 0; j < 3; j++) {
104 if (b[i][j] == '.') {
105 b[i][j] = ai;
106 best = max(best, minimax_misere(b, false, depth + 1));
107 b[i][j] = '.';
108 }
109 }
110 }
111 return best;
112 } else {
113 int best = 1000;
114 for (int i = 0; i < 3; i++) {
115 for (int j = 0; j < 3; j++) {
116 if (b[i][j] == '.') {
117 b[i][j] = opp;
118 best = min(best, minimax_misere(b, true, depth + 1));
119 b[i][j] = '.';
120 }
121 }
122 }
123 return best;
124 }

127pair<int, int> MisereSmartPlayer::calculateMove() {
128 Board<char>* board_ptr = this->get_board_ptr();
129 auto tempBoard = board_ptr->get_board_matrix();
130
131 int bestVal = -1000;
132 int bestX = -1, bestY = -1;
133
134 for (int i = 0; i < 3; i++) {
135 for (int j = 0; j < 3; j++) {
136 if (tempBoard[i][j] == '.') {
137
138 tempBoard[i][j] = symbol;
139
140 int moveVal = minimax_misere(tempBoard, false, 0);
141
142 tempBoard[i][j] = '.';
143
144 if (moveVal > bestVal) {
145 bestVal = moveVal;
146 bestX = i;
147 bestY = j;
148 }
149 }
150 }
151 }
152 return {bestX, bestY};
153}
